cmake_minimum_required(VERSION 3.16)
project(den_of_lords)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Output directories
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib")

# SDL3 configuration
set(CMAKE_PREFIX_PATH "C:/SDL3")
find_package(SDL3 REQUIRED CONFIG)

# GLM
find_package(glm CONFIG REQUIRED)

# Add source files
file(GLOB_RECURSE SOURCES 
    "src/*.cpp" 
) 
add_executable(den_of_lords ${SOURCES})

# Link libraries
target_link_libraries(den_of_lords 
    PRIVATE 
    SDL3::SDL3
    glm::glm
)

# Copy SDL3.dll to output directory
file(GLOB SDL3_DLLS "C:/SDL3/bin/*.dll")
foreach(DLL ${SDL3_DLLS})
    add_custom_command(TARGET den_of_lords  POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
        ${DLL}
        $<TARGET_FILE_DIR:den_of_lords>)
endforeach()
